                             $("#Q4button").on("click", function() {
    var B = ""; //this is a variabale
    var sen = prompt("Just type"); // text box
    var countB = []; //this is an array that counts the B's
    for (var i = 0; i < sen.length; i++) { //i < sen.length; this is the limit of the search (whatever they type in the box)
        if (sen[i] === "B") { //trying to find the B
            countB.push("B"); // push the answer in the screen
        }
    }

    // this text "How many Bs" or if there are no B's text would say "No B's"
    if (countB.length === 0) { // What happens when there are no B's :ALSO .length is number of characters.
        B = "No B's";
    } else { // What happens when there are B's
        B = countB.length;
    }
    $("#Q4result").html(B);
});
